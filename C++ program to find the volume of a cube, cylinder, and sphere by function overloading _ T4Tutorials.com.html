    }
    float T4Tutorials_Volume(int r,int h)
    {
        return(3.14*r*r*h);
    }
    float T4Tutorials_Volume(float r1)
    {
        return((4*3.14*r1*r1*r1)/3);
    }
    int T4Tutorials_Volume(int a)
    {
        return(a*a*a);
    }
